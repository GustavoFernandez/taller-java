Comentarios

/**
* Returns a hash code for this string.
* using {@code int} arithmetic, where {@code s[i]} is the
* ith character of the string, {@code n} is the length of
* the string, and {@code ^} indicates exponentiation.
*/
public int hashCode() {
    int h = hash;
    if (h == 0 && value.length > 0) {
        char val[] = value;
	    /*
	    The hash code for a object is computed as
	    s[0]*31^(n-1) + s[1]*31^(n-2) + ... + s[n-1]
	    */
        for (int i = 0; i < value.length; i++) {
            h = 31 * h + val[i];
        }
        hash = h;
    }
    return h; // return a hash code value for this object.
}

Instrucciones de control

if-else


if (edad >= 18) {
	System.out.println("Mayor de edad!");
}

if (producto.getStock() > 0) {
	efectuarVenta(venta);
	reducirStock(producto, venta.getCantidad());
} else {
	mostrarMensaje("No hay stock: " + producto.getDescription());
}

switch


int mes = 8;
String nombreMes;
switch(mes) {
	case 1:
		nombreMes = "Enero";
		break;
	case 2:
		nombreMes = "Febrero";
		break;
	...
	case 12:
		nombreMes = "Diciembre";
		break;
	default:
		nombreMes = null;
}

while / do-while


int count = 1;
while (count < 11) {
	System.out.println("Count is: " + count);
	count++;
}

int count = 1;
do {
	System.out.println("Count is: " + count);
	count++;
} while (count < 11);

for / for-each


String[] nombres = { "Juan", "Jorge", "Luis", "Rosa" };
for (int i = 0; i < nombres.length; i++) {
	System.out.println(nombres[i]);
}

for (String nombre : nombres) {
	System.out.println(nombre);
}

continue / break


String[] nombres = { "Juan", "Jorge", "Luis", "Rosa" };
// Encontrar índice de algún nombre
String nombreABuscar = "Juan";
int indiceEncontrado = -1;
for (int i = 0; i < nombres.length; i++) {
	if (nombreABuscar.equals(nombres[i])) {
		indiceEncontrado = i;
		break;
	}
}

// No mostrar nombres que inicien en "J"

for (String nombre : nombres) {
	if (nombre.startsWith("J")) {
		continue;
	}
	System.out.println(nombre);
}

Operador ternario

(var = bool expression ? val when true : val when false)


int edad = 12;
String estado = edad >= 18 ? "Mayor de edad" : "Menor de edad";
